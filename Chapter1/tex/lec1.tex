\section{Глава 1. Методы решения некоторых уравнений.}
%06.09.2011

\subsection{Основные понятия}

\textbf{В этом параграфе все величины вещественные.}

\point{1} \textbf{Дифференциальные уравнения первого порядка.}

\Def Обыкновенным дифференциальным уравнением первого порядка называется уравнение
$$
  f(t, y, y') = 0 \eqno (1),
$$
где $t$ -- некоторая переменная, $ y(t) $ -- некоторая заданная функция.

\Def Функция $ y = \varphi(t) $ -- решение $(1)$, если:

\begin{enumerate}
  \item $ y(t) $ определена на некотором промежутке $(\alpha, \beta)$, $[\alpha, \beta]$, $[\alpha, \beta)$, или $(\alpha, \beta]$.
  \item $\varphi'(t)$ непрерывна на $(\alpha, \beta)$, $[\alpha, \beta]$, $[\alpha, \beta)$, или $(\alpha, \beta]$ соответственно.
  \item $ f(t, \varphi(t), \varphi'(t)) = 0 $ на $(\alpha, \beta)$, $[\alpha, \beta]$, $[\alpha, \beta)$, или $(\alpha, \beta]$ соответственно.
\end{enumerate}

\Def Функция
$$
    y = \varphi(t, c) \eqno (2)
$$
где $c$ -- параметр, то есть произвольный коэффициент, -- общее решение уравнения $(1)$, если
\begin{enumerate}
  \item Для любого $c$ функция (2) -- решение,
  \item Конкретное решение (1) может быть получено по общему решению (2) при соответствующем выборе $c$.
\end{enumerate}
Область определения рассматриваемых функций не уточняется.

\point{2} \textbf{Уравнение, разрешенное относительно производной.}

$$
    y' = f(t,y) \eqno (3)
$$
называется \emph{дифференциальным уравнением первого порядка, разрешенным относительно производной.}
$f$ определена в некоторой области $G_{x, y}$ (или замкнутой области).

\Def \emph{Поле направлений} -- множество таких кривых, что график решения огибает поле направлений (касается в каждой точке).
При этом график решения -- не единственный.

\Pic{Chapter1/pic/1.png}

\textbf{Задача Коши для $(3)$:}
\nopagebreak
\begin{itemize}
  \item \textbf{Первая формулировка.} Заданы числа $t_0$, $y_0$, при этом $(t_0, y_0) \in G$; $G$ -- область.
  Найти то решение $y(t)$ уравнения $(3)$, которое удовлетворяет условию
  $$
    y(t_0) = y_0 \eqno (4)
  $$
  \item \textbf{Вторая формулировка.} В области $G$ задана точка. Найти то решение уравнения $(3)$, график которого
  проходит через эту точку.
\end{itemize}

\Th Рассмотрим в области $G$ уравнение $(3)$. Пусть $(t_0, y_0) \in G$. Кроме того, пусть $f$ и $\ud{f}{y}$ непрерывны в $G$.
Тогда:
\begin{enumerate}
  \item Существует решение уравнения $(1)$, удовлетворяющее условию $(4)$ $y(t_0) = y_0$.
  \item Если $y_1(t)$ и $y_2(t)$ -- решение $(3) \land (4)$, то $y_1(t) = y_2(t)$ на пересечении их областей определения.
\end{enumerate}

\point{3} Пусть $y = y(x)$, $x = x(y)$, и рассматривается дифференциальное уравнение вида
$$
    p(x,y) \id x + q(x,t) \id y = 0 \eqno (5),
$$
где $p$ и $q$ определены в $G_{x, y}$, в каждой точке $G$ $p \ne 0$, $q \ne 0$.
Заметим: если на некотором участке $q \ne 0$, то $(5)$ $\sLRA$ $\dfrac{\id y}{\id x} = \dfrac{p(x,y)}{q(x,y)}$.
Аналогично, если $p \ne 0$, то $(5)$ $\sLRA$ $\dfrac{\id x}{\id y} = \dfrac{q(x,y)}{p(x,y)}$.

$(5)$ -- \emph{симметричная форма} дифференциального уравнения первого порядка, разрешённого относительно производной.
Соответствующее решение: $H(x,y) = 0$, общее решение -- $H(x, y, c) = 0$.

\textbf{Задача Коши для (5):}

В $G$ задана точка, найти то решение, график которого проходит через эту точку.

\subsection{Уравнение в полных дифференциалах.}

\textbf{В этом параграфе все величины вещественные.}
$$
    p(x,y)\id x + q(x,y) \id y = 0 \eqno (1),
$$
где $p$ и $q$ -- определены в области $G$, и в любой точке $G$ имеет место $p \ne 0$, $q \ne 0$.

\Def Уравнение $(1)$ -- уравнение в полных дифференциалах, если в $G$ существует функция $U(x,y) \such$
$$
    \id U(x,y) = p(x,y) \id x + q(x,y) \id y.
$$
Если $(5)$ -- уравнение в полных дифференциалах, то
$$
(5) \LRA \id U(x,y) = 0 \LRA U(x,y) = c, \,\, \text{c = const}.
$$

\textbf{Задача Коши:} $x_0$, $y_0$; $c = U(x_0, y_0)$, $U(x,y) = U(x_0, y_0)$.

\textbf{Важные частные случаи (уравнения с разделяющимися переменными):}
$$
    a(x)\id x + b(y) \id y = 0 \eqno (2)
$$
$a(x)$ определена и непрерывна на $(\alpha, \beta)$,
$b(x)$ определена и непрерывна на $(\gamma, \dta)$.
$$
    A(x) \eqdef \int a(x)\id x \text{ -- какая-либо из первообразных на $(\alpha, \beta)$},
$$
$$
    B(x) \eqdef \int b(y)\id y \text{ -- какая-либо из первообразных на $(\gamma, \delta)$}.
$$
$U(x,y) = A(x) + B(y)$. Тогда $\id U(x,y) = A'(x) \id x + B'(y)\id y = a(x) \id x + b(y) \id y$.
$$
    \int a(x) \id x + \int b(y) \id y = c \text{ = const -- решение (2)}
$$
\Rem Получать $U(x,y) = A(x) + B(y)$ интегрированием уравнения $(2)$ нельзя, так как
в уравнении (2) разные слагаемые придется интегрировать по разным переменным, а такой операции нет.
$$
    a(x) m(y) \id x + b(y) n(x) \id y = 0 \eqno (3)
$$
Заметим: $n(\Hat{x}) = 0 \sRA x = \Hat{x}$ -- решение $(3)$, ибо $\id x = 0$, если $\Hat{x}$ = const;
$m(\Hat{y}) = 0 \sRA y = \Hat{y}$ -- решение $(3)$;

Если в некоторой области $m(y) \ne 0$ и $n(x) \ne 0$, то в этой области
$$
    (3) \LRA \dfrac{a(x)}{n(x)}\id x + \dfrac{b(y)}{m(y)} = 0,
$$
то есть уравнение с разделяющимися переменными.

\Task $y = y'$, $G = \{ -\infty < t < +\infty, \,  -\infty < y < +\infty\}$.

$\dfrac{\id y}{\id t} = y$, $\id y - y\id t = 0$, $y \equiv 0$ -- решение.
При $y \ne 0$:
$$
    \dfrac{\id y}{y} - \id t = 0, \quad \mathrm{ln} (y) - t = c, \quad y = \pm e^c e^t,
$$
$\pm e^c$ -- произвольная ненулевая константа.
Общее решение: $y = D e^t$, $D$ -- произвольная константа.

Если заданы $t_0$, $y_0$, то $y_0 = D e^{t_0} \RA D = y_0 e^{-t_0}$.
$$
    y(t) = y_0 e^{t - t_0} \text{ -- решение задачи Коши.}
$$

\Task $y' = \sqrt[3]{y^2}$, $G = \{ -\infty < t < +\infty, \,  -\infty < y < +\infty\}$.
$$
    \dfrac{\id y}{\id t} = \sqrt[3]{y^2}; \quad \id y - \sqrt[3]{2}\id t = 0.
$$
$y \equiv 0$ -- решение. Если $y \ne 0$:
$$
    \dfrac{\id y}{\sqrt[3]{y^2}} - \id t = 0, \quad \dfrac{3y}{\sqrt[3]{y^2}} - t = c; \quad
    y = \left( \dfrac{t+c}{3} \right)^3.
$$
Эта функция удовлетворяет уравнению и при $y = 0$.
